#include "../libraries/Print.hpp"

template <typename T>
void println(T& message,  Color& color) {
    cout << COLORS[static_cast<int>(color)] << message << RESET << endl;
}

template <typename T>
void println(T& message) {
    cout << message << endl;
}

template <typename T>
void print( T& message,  Color& color) {
    cout << COLORS[static_cast<int>(color)] << message << RESET;
}

template <typename T>
void print( T& message) {
    cout << message;
}

template <typename T>
string convertString( T& variable) {
    ostringstream oss;
    oss << variable;
    return oss.str();
}
